Check Product Authenticity

function doGet(e) { const code = e.parameter.code; const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("ProductCodes"); if (!sheet) { return ContentService.createTextOutput(JSON.stringify({ valid: false, error: "Sheet 'ProductCodes' not found" })).setMimeType(ContentService.MimeType.JSON); } const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++){ if (data[i][0] === code) { return ContentService .createTextOutput(JSON.stringify({ valid: true, product: data[i][1], status: data[i][2] })) .setMimeType(ContentService.MimeType.JSON); } } return ContentService .createTextOutput(JSON.stringify({ valid: false })) .setMimeType(ContentService.MimeType.JSON); }
